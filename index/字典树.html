字典树

2024/2/26 
思路：现在想象一个平行线，上面有a~z字母，每个占一格，从左到右排列
每一个字母的下方接着排列a~z字母。
存放的时候我们对着出现的字母依次往下放（点亮格子）
如果是第一次出现顺便给他标号,如f[10]['a']=11，标号为10的下一层的a儿子标号为11
此处'a'可以映射为别的值，不然太占内存。
（第一个平行线的父亲是一个虚拟源点，不然不好标号），可以默认虚拟源点的下标为0

用途广泛，需要自行考虑是否用到

下面用到的是查询某个字符串t 在一大堆字符串s中，有多少个s是t的前缀（大小写都有）

本题思路：构造s的字典树，在每个独立的字符串最后一个字符上+1（此字符串出现一次）
用t遍历s的字典树，取路上最大的值（即总共有多少个字符串可以为前缀）

#include <bits/stdc++.h>
//#define int long long
#define per(i,j,k) for(int (i)=(j);(i)<=(k);++(i))
#define rep(i,j,k) for(int (i)=(j);(i)>=(k);--(i))
#define fr first
#define se second
#define endl '\n'
using namespace std;
const int N=3e6+5;

int T,n,q,t[N][62],cnt[N],idx;
string s;

int num(char c){
    if(c>='0' and c<='9')return c-'0';//0~9
    else if(c>='A' and c<='Z')return c-'A'+10;//10~35
    else return c-'a'+36;//36~61
}

void solve(){
    cin>>n>>q;
    per(i,1,n){
        cin>>s;
        int now=0,nxt;
        per(j,0,s.length()-1){//构建字典树
            nxt=num(s[j]);
            if(!t[now][nxt])t[now][nxt]=++idx;
            now=t[now][nxt];
            cnt[now]++;
        }
    }
    per(i,1,q){
        cin>>s;
        int now=0,j=0;
        while(t[now][num(s[j])] and j<s.length())now=t[now][num(s[j++])];
        cout<<((j==s.length())?cnt[now]:0)<<endl;
    }
}

void init(){
    per(i,0,idx){
        cnt[i]=0;
        per(j,0,61)t[i][j]=0;
    }
    idx=0;
}
signed main(){
    ios::sync_with_stdio(false),cin.tie(nullptr);
    int t;
    cin>>t;
    while(t--)solve(),init();
    return 0;
}
