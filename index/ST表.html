ST表

2024/2/26

假设N个元素，Q次询问
ST表复杂度：NlogN+Q
线段树复杂度：N+NlogQ
根据复杂度可知，一般情况下可以使用线段树来解决ST表问题，除非真的很卡时间。

所以下面直接使用线段树

适用范围：多次访问 静态区间 的 极值 问题

线段树的下标规律：点x的左儿子为x*2，右儿子x*2+1，即x<<1和x<<1|1（左移一位之后必然会空出最低位x<<1|1是合法的）

线段树开4倍内存

建树：从根节点开始往下左右分掌控区间，后序遍历处理内容（如区间最大值，区间和）

访问：从[1,n]总范围开始递归，如果递归到的区间满足在[x,y]区间内，就返回值

快读：在洛谷用优化的cin和cout会超时，要用快读和printf

懒标记：用到修改才需要使用，本题不需要

#include <bits/stdc++.h>
//#define int long long
#define per(i,j,k) for(int (i)=(j);(i)<=(k);++(i))
#define rep(i,j,k) for(int (i)=(j);(i)>=(k);--(i))
#define fr first
#define se second
#define endl '\n'
using namespace std;
const int N=1e5+5;

int n,m,a[N],t[N<<2],tag[N<<2],x,y;

int read(){
    int x=0,f=1;char ch=getchar();
    while(ch<'0' or ch>'9'){
        if(ch=='-')f=-1;
        ch=getchar();
    }
    while(ch>='0' and ch<='9'){
        x=(x<<3)+(x<<1)+(ch xor 48);
        ch=getchar();
    }
    return x*f;
}

int lc(int x){
    return x<<1;
}

int rc(int x){
    return x<<1|1;
}

void update(int i){
    t[i]=max(t[lc(i)],t[rc(i)]);
}

void build(int l,int r,int i){
    if(l==r){
        t[i]=a[l];
        return;
    }
    int mid=(l+r)>>1;
    build(l,mid,lc(i));
    build(mid+1,r,rc(i));
    update(i);
}

int query(int l,int r,int i,int x,int y){
    int ans=0;
    if(l>=x and r<=y)return t[i];
    int mid=(l+r)>>1;
    if(x<=mid)ans=max(ans,query(l,mid,lc(i),x,y));
    if(y>mid)ans=max(ans,query(mid+1,r,rc(i),x,y));
    return ans;
}

void solve(){
    n=read(),m=read();
    per(i,1,n)a[i]=read();

    build(1,n,1);

    per(i,1,m){
        x=read(),y=read();
        printf("%d\n", query(1,n,1,x,y));
    }
}

signed main(){
    ios::sync_with_stdio(false),cin.tie(nullptr);
    int t=1;
    while(t--)solve();
    return 0;
}
